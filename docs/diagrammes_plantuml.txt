@startuml DiagrammeClasses
skinparam classBackgroundColor LightYellow
skinparam classBorderColor DarkBlue

class Utilisateur {
  - id: Long
  - nom: String
  - prenom: String
  - email: String
  - motDePasse: String
  - role: Role
  - actif: Boolean
  + seConnecter(): boolean
  + getNomComplet(): String
}

class Ouvrage {
  - id: Long
  - titre: String
  - isbn: String
  - anneePublication: int
  - nbExemplaires: int
  - nbDisponibles: int
  + estDisponible(): boolean
  + decrementerDisponibles(): void
  + incrementerDisponibles(): void
}

class Auteur {
  - id: Long
  - nom: String
  - prenom: String
  - biographie: String
}

class Categorie {
  - id: Long
  - nom: String
  - rayon: String
  - description: String
}

class Exemplaire {
  - id: Long
  - cote: String
  - etat: EtatExemplaire
  - disponible: boolean
}

class Emprunt {
  - id: Long
  - dateEmprunt: LocalDate
  - dateRetourPrevue: LocalDate
  - dateRetourEffective: LocalDate
  - statut: StatutEmprunt
  + estEnRetard(): boolean
  + calculerJoursRetard(): long
  + calculerPenalite(): double
}

class Reservation {
  - id: Long
  - dateReservation: LocalDateTime
  - statut: StatutReservation
  - dateExpiration: LocalDate
}

class Penalite {
  - id: Long
  - montant: double
  - dateCalcul: LocalDate
  - payee: boolean
  - joursRetard: long
}

enum Role { ETUDIANT, BIBLIOTHECAIRE, ADMINISTRATEUR }
enum StatutEmprunt { EN_COURS, RETOURNE, EN_RETARD }
enum StatutReservation { ACTIVE, SATISFAITE, EXPIREE, ANNULEE }
enum EtatExemplaire { BON, ABIME, PERDU }

Utilisateur --> Role
Utilisateur "1" --> "*" Emprunt : effectue
Utilisateur "1" --> "*" Reservation : crée
Utilisateur "1" --> "*" Penalite : accumule

Ouvrage "*" --> "*" Auteur : écrit par
Ouvrage "*" --> "1" Categorie : appartient
Ouvrage "1" --> "*" Exemplaire : possède
Ouvrage "1" --> "*" Reservation

Exemplaire "1" --> "*" Emprunt : fait objet
Emprunt "1" --> "0..1" Penalite : génère
Emprunt --> StatutEmprunt
Reservation --> StatutReservation
Exemplaire --> EtatExemplaire
@enduml

@startuml DiagrammeSequenceEmprunt
actor Etudiant
participant "Interface Web" as UI
participant "EmpruntController" as EC
participant "EmpruntService" as ES
participant "OuvrageRepository" as OR
participant "ExemplaireRepository" as ExR
participant "EmpruntRepository" as ER
participant "NotificationService" as NS

Etudiant -> UI : Clique sur 'Emprunter'
UI -> EC : POST /api/emprunts?ouvrageId=1&etudiantId=5
EC -> ES : creerEmprunt(1, 5)
ES -> OR : findById(1)
OR --> ES : Ouvrage{titre, nbDisponibles=1}
ES -> ES : verifier estDisponible()

alt Ouvrage disponible
  ES -> ES : verifier penalites etudiant
  ES -> ExR : findByOuvrageIdAndDisponibleTrue(1)
  ExR --> ES : Exemplaire{cote='INF-001-A'}
  ES -> OR : save(ouvrage - nbDisponibles)
  ES -> ExR : save(exemplaire.disponible=false)
  ES -> ER : save(newEmprunt)
  ER --> ES : Emprunt{id=42, dateRetour=J+14}
  ES -> NS : planifierRappelRetour(emprunt)
  NS --> ES : email envoyé
  ES --> EC : Emprunt DTO
  EC --> UI : 201 Created + EmpruntDTO
  UI --> Etudiant : ✅ Emprunt enregistré - Retour le [date]

else Ouvrage indisponible
  ES --> EC : RuntimeException("indisponible")
  EC --> UI : 409 Conflict
  UI --> Etudiant : ❌ Indisponible - Voulez-vous réserver ?
end
@enduml
